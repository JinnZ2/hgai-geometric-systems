    "def calculate_m_s(metrics: EducationMetrics) -> float:\n",
    "    \"\"\"Calculate M(S) = (R_e Ã— A Ã— D Ã— C) - L\"\"\"\n",
    "    coherence = (\n",
    "        metrics.resonance * \n",
    "        metrics.adaptability * \n",
    "        metrics.diversity * \n",
    "        metrics.curiosity\n",
    "    )\n",
    "    return coherence - metrics.loss\n",
    "\n",
    "def interpret_m_s(m_s: float) -> str:\n",
    "    \"\"\"Interpret M(S) score for education\"\"\"\n",
    "    if m_s > 7:\n",
    "        return \"Highly effective, produces lifelong learners\"\n",
    "    elif m_s > 5:\n",
    "        return \"Strong learning environment, engaged students\"\n",
    "    elif m_s > 3:\n",
    "        return \"Adequate learning, mixed engagement\"\n",
    "    elif m_s > 1:\n",
    "        return \"Struggling, requires external motivation\"\n",
    "    elif m_s > 0:\n",
    "        return \"Dysfunctional, high disengagement\"\n",
    "    else:\n",
    "        return \"Harmful, destroys love of learning\"\n",
    "\n",
    "# Calculate M(S) for all approaches\n",
    "results = []\n",
    "for school in schools:\n",
    "    m_s = calculate_m_s(school)\n",
    "    results.append({\n",
    "        'Educational Approach': school.name,\n",
    "        'R_e': f\"{school.resonance:.2f}\",\n",
    "        'A': f\"{school.adaptability:.2f}\",\n",
    "        'D': f\"{school.diversity:.2f}\",\n",
    "        'C': f\"{school.curiosity:.2f}\",\n",
    "        'L': f\"{school.loss:.2f}\",\n",
    "        'M(S)': f\"{m_s:.2f}\",\n",
    "        'Assessment': interpret_m_s(m_s)\n",
    "    })\n",

    "def simulate_student_trajectory(system_type: str, years: int = 12) -> List[Dict]:\n",
    "    \"\"\"\n",
    "    Simulate student engagement and learning retention over time\n",
    "    \"\"\"\n",
    "    timeline = []\n",
    "    \n",
    "    # Get system M(S)\n",
    "    system_map = {\n",
    "        'curiosity': self_directed,\n",
    "        'standardized': standardized_system,\n",
    "        'test_prep': test_prep_academy\n",
    "    }\n",
    "    system = system_map[system_type]\n",
    "    base_m_s = calculate_m_s(system)\n",
    "    \n",
    "    # Starting enthusiasm (all students start curious)\n",
    "    initial_enthusiasm = 0.90\n",
    "    \n",
    "    for year in range(years + 1):\n",
    "        # Enthusiasm changes based on M(S)\n",
    "        if base_m_s > 5:\n",
    "            # High M(S): enthusiasm maintained or grows\n",
    "            enthusiasm = min(0.98, initial_enthusiasm + year * 0.005)\n",
    "            knowledge_retention = 0.85 + enthusiasm * 0.10\n",
    "        elif base_m_s > 0:\n",
    "            # Medium M(S): gradual decline\n",
    "            decline_rate = 0.03 * (5 - base_m_s) / 5\n",
    "            enthusiasm = max(0.30, initial_enthusiasm - year * decline_rate)\n",
    "            knowledge_retention = 0.40 + enthusiasm * 0.30\n",
    "        else:\n",
    "            # Negative M(S): rapid collapse\n",
    "            decline_rate = 0.08\n",
    "            enthusiasm = max(0.10, initial_enthusiasm - year * decline_rate)\n",
    "            knowledge_retention = 0.15 + enthusiasm * 0.15\n",
    "        \n",
    "        timeline.append({\n",
    "            'year': year,\n",
    "            'enthusiasm': enthusiasm,\n",
    "            'knowledge_retention': knowledge_retention,\n",
    "            'system': system_type\n",
    "        })\n",
    "    \n",
    "    return timeline\n",
