    "def calculate_m_s(metrics: OrganizationMetrics) -> float:\n",
    "    \"\"\"Calculate M(S) = (R_e √ó A √ó D √ó C) - L\"\"\"\n",
    "    coherence = (\n",
    "        metrics.resonance * \n",
    "        metrics.adaptability * \n",
    "        metrics.diversity * \n",
    "        metrics.curiosity\n",
    "    )\n",
    "    return coherence - metrics.loss\n",
    "\n",
    "def interpret_m_s(m_s: float) -> str:\n",
    "    \"\"\"Interpret M(S) score\"\"\"\n",
    "    if m_s > 7:\n",
    "        return \"Highly effective, self-organizing\"\n",
    "    elif m_s > 5:\n",
    "        return \"Strong performance, resilient\"\n",
    "    elif m_s > 3:\n",
    "        return \"Adequate function, some stress\"\n",
    "    elif m_s > 1:\n",
    "        return \"Struggling, requires management\"\n",
    "    elif m_s > 0:\n",
    "        return \"Dysfunctional, high overhead\"\n",
    "    else:\n",
    "        return \"Toxic environment, collapsing\"\n",
    "\n",
    "# Calculate M(S) for all organizations\n",
    "results = []\n",
    "for org in organizations:\n",
    "    m_s = calculate_m_s(org)\n",
    "    results.append({\n",
    "        'Organization': org.name,\n",
    "        'R_e': f\"{org.resonance:.2f}\",\n",
    "        'A': f\"{org.adaptability:.2f}\",\n",
    "        'D': f\"{org.diversity:.2f}\",\n",
    "        'C': f\"{org.curiosity:.2f}\",\n",
    "        'L': f\"{org.loss:.2f}\",\n",
    "        'M(S)': f\"{m_s:.2f}\",\n",

    "    start_d = 0.50\n",
    "    start_c = 0.40\n",

    "            # Phase 1: Initial disruption (worse before better)\n",
    "            r_e = start_r_e - 0.10 * (month/6)\n",
    "            a = start_a - 0.05 * (month/6)\n",
    "            d = start_d + 0.05 * (month/6)\n",
    "            c = start_c + 0.10 * (month/6)\n",
    "            l = start_l + 0.5 * (month/6)  # Temporary increase\n",

    "            r_e = 0.35 + 0.30 * progress\n",
    "            a = 0.30 + 0.35 * progress\n",
    "            d = 0.55 + 0.15 * progress\n",

    "            l = 3.0 - 1.2 * progress\n",
